def deleteNewsArchive(path){
    def pageMgr = slingRequest.getResourceResolver().adaptTo(com.day.cq.wcm.api.PageManager)
    getPage(path).recurse { page ->
      if(!page.path.endsWith("about-tetra-pak/news-and-events/newsarchive")){
          deactivate page.path
          pageMgr.delete(page,false,true)
      }
    }
}
